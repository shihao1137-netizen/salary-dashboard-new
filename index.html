function doGet(e) { try { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Raw data"); if (!sheet) throw new Error("找不到工作表：Raw data"); const action = e && e.parameter && e.parameter.action ? String(e.parameter.action).trim() : ""; const callback = e && e.parameter && e.parameter.callback ? String(e.parameter.callback).trim() : ""; // ===== 寫入資料 ===== if (action === "append") { const year = Number(e.parameter.year); const month = Number(e.parameter.month); const type = String(e.parameter.type || "").trim(); const category = String(e.parameter.category || "").trim(); let amount = Number(e.parameter.amount || 0); const note = String(e.parameter.note || "").trim(); if (!year || !month || !type || !category || !amount) { return outputJsonp(callback, { success: false, error: "欄位不完整" }); } if (type === "支出") { amount = -Math.abs(amount); } else { amount = Math.abs(amount); } const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0].map(h => String(h).trim()); const noteColIndex = headers.findIndex(h => h === "note" || h === "備註"); const newRow = new Array(Math.max(sheet.getLastColumn(), 5)).fill(""); newRow[0] = year; newRow[1] = month; newRow[2] = type; newRow[3] = category; newRow[4] = amount; if (noteColIndex !== -1) { newRow[noteColIndex] = note; } sheet.appendRow(newRow); return outputJsonp(callback, { success: true, message: "新增成功" }); } // ===== 讀取資料 ===== const data = sheet.getDataRange().getValues(); const result = []; for (let i = 1; i < data.length; i++) { const row = data[i]; if (!row[0] || !row[1] || !row[2] || !row[3] || row[4] === "") continue; result.push({ year: Number(row[0]), month: Number(row[1]), type: String(row[2]).trim(), category: String(row[3]).trim(), amount: Number(row[4]) }); } return outputJsonp(callback, result); } catch (error) { return outputJsonp( e && e.parameter && e.parameter.callback ? String(e.parameter.callback) : "", { success: false, error: String(error) } ); } } function outputJsonp(callback, data) { const json = JSON.stringify(data); if (callback) { return ContentService .createTextOutput(callback + "(" + json + ")") .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); }
